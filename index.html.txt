  const a=dayStops(); if(i<0||j<0||i>=a.length||j>=a.length) return;

function moveStop(from,to){
  const a=dayStops(); if(from<0||from>=a.length) return;

}

function timeAdd(hhmm,mins){ const m=/^(\d{1,2}):(\d{2})$/.exec(hhmm); if(!m) return '09:00'; let t=(+m[1])*60+(+m[2])+mins; t=((t%1440)+1440)%1440; return String(Math.floor(t/60)).padStart(2,'0')+':'+String(t%60).padStart(2,'0'); }

/* 營業時間檢查：抵達時間是否在 open–close 內（close 可跨午夜，例 00:00 視為次日） */
function toMin(t){ const m=/^(\d{1,2}):(\d{2})$/.exec(t||''); return m?(+m[1])*60+(+m[2]):null; }
function fromMin(x){ x=((x%1440)+1440)%1440; return String(Math.floor(x/60)).padStart(2,'0')+':'+String(x%60).padStart(2,'0'); }
function hoursWarning(s){
  if(!s||s.isAnchor||!s.open||!s.close) return null;
  const o=toMin(s.open); let c=toMin(s.close); if(o==null||c==null) return null; if(c<=o) c+=1440;
  let a=toMin(s.time); if(a==null) return null; if(a<o && a+1440<=c) a+=1440;
  if(a<o||a>c) return '抵達時間 '+s.time+' 不在營業時間 '+s.open+'–'+s.close+' 內';
  const dur=parseInt((String(s.dur||'').match(/\d+/)||[0])[0])||0;
  if(dur && a+dur>c) return '預估停留至 '+fromMin(a+dur)+'，可能超過營業結束 '+s.close;
  return null;
}

function markActivePreview(id){ activeId=id; renderMap(); renderItinerary(); renderAlternatives(); }
function preview(){
  if(!lmap) return;
  const pts=dayCoordPoints(); if(previewing||pts.length<2) return;
  previewing=true; hideDetail();
  const btn=$('#previewBtn'); btn.disabled=true; btn.classList.add('on');
  btn.innerHTML='<svg viewBox="0 0 24 24" fill="currentColor"><rect x="6" y="5" width="4" height="14" rx="1"/><rect x="14" y="5" width="4" height="14" rx="1"/></svg>預走中…';
  travelerMarker=L.circleMarker([pts[0].lat,pts[0].lng],{radius:8,color:'#fff',weight:3,fillColor:'#EE6A3C',fillOpacity:1}).addTo(lmap);
  let i=0; const SEG=1200,PAUSE=650;
  markActivePreview(pts[0].id);
  zoomLeg();
  function zoomLeg(){
    if(i>=pts.length-1){ lmap.flyToBounds(L.latLngBounds(pts.map(p=>[p.lat,p.lng])).pad(0.25),{duration:0.8,maxZoom:16}); setTimeout(finish,950); return; }
    const a=pts[i],b=pts[i+1];
    lmap.flyToBounds(L.latLngBounds([[a.lat,a.lng],[b.lat,b.lng]]).pad(0.55),{duration:0.7,maxZoom:17});
    setTimeout(()=>animateLeg(a,b),780);
  }
  function animateLeg(a,b){ const t0=performance.now();
    (function fr(now){ const e=smoothstep(Math.min(1,(now-t0)/SEG)); travelerMarker.setLatLng([a.lat+(b.lat-a.lat)*e, a.lng+(b.lng-a.lng)*e]);
      if((now-t0)<SEG) requestAnimationFrame(fr); else { i++; markActivePreview(pts[i].id); setTimeout(zoomLeg,PAUSE); } })(t0);

/* ============ 儲存 / 分享 / 匯出匯入 ============
   注意：localStorage 在 claude.ai 的預覽沙箱會被擋（已用 try/catch 包住，不會報錯）；
   一旦把這個 HTML 自行發佈到網站（見對話說明），儲存與分享連結就會完整運作。 */
const LS_KEY='trip-planner-state-v2';
function normalizeOptions(){ for(const c in CATS){ if(!options[c]) options[c]=[]; } }
function snapshot(){ return {v:1, title:trip.title, library, options, trip, activeDay}; }
function applyState(s){ if(!s||!s.trip||!s.library) return false; library=s.library; options=s.options||{}; trip=s.trip; if(s.title) trip.title=s.title; normalizeOptions(); activeDay=Math.max(0,Math.min(s.activeDay||0,(trip.days.length||1)-1)); return true; }
function saveState(){ try{ localStorage.setItem(LS_KEY, JSON.stringify(snapshot())); }catch(e){} }
function loadState(){ try{ const r=localStorage.getItem(LS_KEY); return r? applyState(JSON.parse(r)) : false; }catch(e){ return false; } }
let _pt; function schedulePersist(){ if(previewing) return; clearTimeout(_pt); _pt=setTimeout(saveState,400); }

function encodeState(){ return btoa(unescape(encodeURIComponent(JSON.stringify(snapshot())))); }
function tryLoadFromHash(){ try{ const m=/[#&]trip=([^&]+)/.exec(location.hash||''); if(!m) return false; const ok=applyState(JSON.parse(decodeURIComponent(escape(atob(m[1]))))); if(ok){ try{ history.replaceState(null,'',location.pathname+location.search); }catch(e){} } return ok; }catch(e){ return false; } }
function doShare(){
  let url; try{ url=location.origin+location.pathname+'#trip='+encodeState(); }catch(e){ url=location.href; }
  const ok=()=>toast('已複製行程連結，貼給朋友即可開啟這份行程');
  if(navigator.clipboard&&navigator.clipboard.writeText) navigator.clipboard.writeText(url).then(ok,()=>prompt('複製此連結分享：',url));
  else prompt('複製此連結分享：',url);
}
function renderAll(){ renderDayTabs(); renderMap(); renderItinerary(); renderAlternatives(); }
function exportJSON(){
  try{ const blob=new Blob([JSON.stringify(snapshot(),null,2)],{type:'application/json'});
    const a=document.createElement('a'); a.href=URL.createObjectURL(blob); a.download=(trip.title||'trip').replace(/\s+/g,'_')+'.json'; document.body.appendChild(a); a.click(); a.remove(); setTimeout(()=>URL.revokeObjectURL(a.href),2000); toast('已匯出 JSON 檔'); }catch(e){ toast('匯出失敗'); }
}
function importJSON(file){ const r=new FileReader(); r.onload=()=>{ try{ const s=JSON.parse(r.result); if(applyState(s)){ activeId=null; renderAll(); saveState(); closeLib(); toast('已匯入並載入行程'); } else toast('檔案格式不符'); }catch(e){ toast('匯入失敗：無法解析檔案'); } }; r.readAsText(file); }

$('#shareBtn').addEventListener('click',doShare);
$('#exportBtn').addEventListener('click',exportJSON);

$('#importFile').addEventListener('change',e=>{ const f=e.target.files&&e.target.files[0]; if(f) importJSON(f); e.target.value=''; });
